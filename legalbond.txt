        let pdfDoc = null;
        let currentPage = 1;
        let canvas = null;
        let ctx = null;
        let pdfLoaded = false;
        
        async function renderPDF(file) {
            try {
                const fileReader = new FileReader();
                fileReader.onload = async function() {
                    const typedarray = new Uint8Array(this.result);
                    pdfDoc = await pdfjsLib.getDocument(typedarray).promise;
                    
                    pdfLoaded = true;
                    document.getElementById('annotationToolbar').style.display = 'flex';
                    
                    // Render first page
                    await renderPage(1);
                    
                    ethanMessage(`Loaded PDF with ${pdfDoc.numPages} pages.`);
                };
                fileReader.readAsArrayBuffer(file);
            } catch (error) {
                console.error('PDF loading error:', error);
                ethanMessage('Error loading PDF. Please try another file.');
            }
        }
        
        async function renderPage(pageNum) {
            if (!pdfDoc) return;
            
            const page = await pdfDoc.getPage(pageNum);
            const scale = 1.5;
            const viewport = page.getViewport({ scale });
            
            // Create or clear canvas
            const viewer = document.getElementById('pdfViewer');
            
            if (!canvas) {
                canvas = document.createElement('canvas');
                canvas.className = 'pdf-canvas';
                viewer.innerHTML = '';
                viewer.appendChild(canvas);
            }
            
            ctx = canvas.getContext('2d');
            canvas.height = viewport.height;
            canvas.width = viewport.width;
            
            const renderContext = {
                canvasContext: ctx,
                viewport: viewport
            };
            
            await page.render(renderContext).promise;
            currentPage = pageNum;
        }

        // Ethan command processor
        function processEthanCommand(input) {
            const lowerInput = input.toLowerCase().trim();
            
            if (lowerInput.startsWith('/')) {
                const command = lowerInput.substring(1).split(' ')[0];
                
                const commands = {
                    'help': () => {
                        return `Available commands:\n\n/highlight - Activate highlight tool\n/pen - Activate pen tool\n/eraser - Activate eraser tool\n/bookmark - Add bookmark\n/summarize - Summarize current page\n/search [term] - Search statute library\n/next - Next page\n/prev - Previous page`;
                    },
                    'highlight': () => {
                        document.getElementById('highlightBtn').click();
                        return 'Highlight tool activated.';
                    },
                    'pen': () => {
                        document.getElementById('penRedBtn').click();
                        return 'Pen tool activated.';
                    },
                    'eraser': () => {
                        document.getElementById('eraserBtn').click();
                        return 'Eraser tool activated.';
                    },
                    'bookmark': () => {
                        return `Page ${currentPage} bookmarked.`;
                    },
                    'summarize': () => {
                        return pdfLoaded ? `Summary of page ${currentPage}: This feature requires GPT/Gemini API integration. Add your API key to enable AI-powered summarization.` : 'No PDF loaded.';
                    },
                    'next': () => {
                        if (pdfLoaded && currentPage < pdfDoc.numPages) {
                            renderPage(currentPage + 1);
                            return `Navigated to page ${currentPage + 1}.`;
                        }
                        return 'No next page.';
                    },
                    'prev': () => {
                        if (pdfLoaded && currentPage > 1) {
                            renderPage(currentPage - 1);
                            return `Navigated to page ${currentPage - 1}.`;
                        }
                        return 'No previous page.';
                    }
                };
                
                if (commands[command]) {
                    return commands[command]();
                } else if (command === 'search') {
                    const searchTerm = input.substring(8);
                    return `Searching statute library for "${searchTerm}"...`;
                } else {
                    return `Unknown command: /${command}. Type /help for available commands.`;
                }
            } else {
                // Natural language processing (placeholder for API integration)
                return `I received your message: "${input}". To enable AI responses, integrate with OpenAI or Google Gemini API. For now, use commands like /help, /highlight, /summarize, etc.`;
            }
        }
